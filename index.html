function searchRaces() {

const searchTerm = document.getElementById('searchInput').value.toLowerCase();

const raceCards = document.querySelectorAll('.race-card');

raceCards.forEach(card => {

const title = card.querySelector('.race-title').textContent.toLowerCase();

const description = card.querySelector('.race-description').textContent.toLowerCase();

const origin = card.querySelector('.race-origin').textContent.toLowerCase();

if (title.includes(searchTerm) || description.includes(searchTerm) || origin.includes(searchTerm)) {

card.style.display = 'block';

} else {

card.style.display = 'none';

}

});

}
